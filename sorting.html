			/**
			 * Partition method used in QUICK-SORT
			 * Implementation.
			 * @param a
			 * @param p
			 * @param r
			 * @return
			 */
			static int partition(Integer [] a, int p, int r) {
				int pivot = a[r];
				int i = p - 1;
				for(int j = p; j <= r-1; j++) {
					if(a[j] <= pivot) {
						i = i + 1;
						int temp = a[j];
						a[j] = a[i];
						a[i] = temp;
					}
				}
				int temp = a[r];
				a[r] = a[i+1];
				a[i+1] = temp;
				return i+1;
			}
			
			/** 
			 * Implementation of QUICK-SORT Sorting algorithm.
			 * Uses the partition method defined above.
			 * @param a
			 * @param p
			 * @param r
			 */
			public static void sort(Integer [] a, int p, int r) {
				if (p < r) {
					int q = partition(a,p,r);
					sort(a,p,q-1);
					sort(a,q+1,r);
				}
			}
