			/**
			 * Partition method used in QUICK-SORT
			 * Implementation.
			 * @param a
			 * @param p
			 * @param r
			 * @return
			 */
			static int partition(Integer [] a, int p, int r) {
				int pivot = a[r];
				int i = p - 1;
				for(int j = p; j <= r-1; j++) {
					if(a[j] <= pivot) {
						i = i + 1;
						int temp = a[j];
						a[j] = a[i];
						a[i] = temp;
					}
				}
				int temp = a[r];
				a[r] = a[i+1];
				a[i+1] = temp;
				return i+1;
			}
			
			/** 
			 * Implementation of QUICK-SORT Sorting algorithm.
			 * Uses the partition method defined above.
			 * @param a
			 * @param p
			 * @param r
			 */
			public static void sort(Integer [] a, int p, int r) {
				if (p < r) {
					int q = partition(a,p,r);
					sort(a,p,q-1);
					sort(a,q+1,r);
				}
			}

				}

			/**
			 * Variation of QUICK-SORT in which the median value
			 * of the array is chosen as the initial pivot.
			 * @param a
			 * @param p
			 * @param r
			 */
			public static void sortMedian(Integer [] a, int p, int r) {
				getMedian(a, p, r);
				if (p < r) {
					int q = partition(a,p,r);
					sort(a,p,q-1);
					sort(a,q+1,r);
				}
			}
=> Quick-Sort implementation with 3 partition split 
		
			/**
			 * 1d. Partition method to use in 3wary partition scheme
			 * QUICK-SORT
			 * @param a
			 * @param p
			 * @param r
			 * @return
			 */
			static int [] partition3Way(Integer [] a, int p, int r) {
				int lessThan = p;
				int i;
				int greaterThan = r;
				int pivot = a[p];
				
				for(i = p; i <= greaterThan; i++) {
					if(a[i] < pivot) {
						int temp = a[lessThan];
						a[lessThan] = a[i];
						a[i] = temp;
						lessThan += 1;
						i += 1;				
					}
					
					else if(a[i] > pivot) {
						int temp = a[i];
						a[i] = a[greaterThan];
						a[greaterThan] = temp;
						greaterThan -= 1;
					}
					
					else {
						i += 1;
					}
				}
				int [] threeWay = {lessThan, greaterThan};
				return threeWay;
			}
			
			/**
			 * 1d. Variation of QUICK-SORT in which an array is
			 * divided in 3 parts. Good for repetition of 
			 * duplicates.
			 * @param a
			 * @param p
			 * @param r
			 */
			public static void sort3Way(Integer [] a, int p, int r) {
				if (p < r) {
					int i = partition3Way(a,p,r)[0];
					int j = partition3Way(a,p,r)[1];
					sort3Way(a,p,i);
					sort3Way(a,j,r);
				}
